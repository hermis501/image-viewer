# capture image using computer's camera
import cv2
webcam = cv2.VideoCapture(0)

#function
#param: filename, for instance: c:/file.jpg

def capture (param):
	try:
		check, frame = webcam.read()
		cv2.imwrite(filename=param, img=frame)
	except:
		pass

# release webcam object before closing the program
webcam.release()
